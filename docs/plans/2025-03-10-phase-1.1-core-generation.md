```typescript
// backend/src/services/ai/base.service.ts
import { TextGenerationRequest, TextGenerationResult, CostEstimate, AIServiceError } from '../../types/ai.types';
import { logger } from '../../utils/logger';

export abstract class BaseAIService {
  protected apiKey: string;
  protected baseUrl: string;

  constructor(apiKey: string, baseUrl: string) {
    this.apiKey = apiKey;
    this.baseUrl = baseUrl;
  }

  // Abstract methods to be implemented by each service
  abstract generateText(request: TextGenerationRequest): Promise<TextGenerationResult>;
  abstract estimateCost(prompt: string, options: any): Promise<CostEstimate>;
  abstract validateApiKey(): Promise<boolean>;

  // Common utility methods
  protected handleError(error: any, provider: string): AIServiceError {
    logger.error(`AI Service Error (${provider}):`, error);

    // Determine if error is retryable
    const retryableErrors = ['RATE_LIMIT', 'TIMEOUT', 'SERVER_ERROR'];
    const isRetryable = retryableErrors.some(code =>
      error.code?.includes(code) || error.message?.includes(code)
    );

    return {
      code: error.code || 'UNKNOWN_ERROR',
      message: error.message || 'An unknown error occurred',
      provider: provider as any,
      retryable: isRetryable
    };
  }

  protected calculateCost(inputTokens: number, outputTokens: number, pricing: any): number {
    return (inputTokens * pricing.input) + (outputTokens * pricing.output);
  }
}
```

```typescript
// backend/src/services/ai/glm.service.ts
import axios, { AxiosError } from 'axios';
import { BaseAIService } from './base.service';
import {
  TextGenerationRequest,
  TextGenerationResult,
  CostEstimate,
  AIProvider,
  AIModel
} from '../../types/ai.types';
import { logger } from '../../utils/logger';

export class GLMService extends BaseAIService {
  private readonly pricing = {
    input: 0.0001,  // ¥0.0001 per token
    output: 0.0002  // ¥0.0002 per token
  };

  async generateText(request: TextGenerationRequest): Promise<TextGenerationResult> {
    try {
      logger.info(`GLM generating text for prompt: ${request.prompt.substring(0, 50)}...`);

      const response = await axios.post(
        `${this.baseUrl}chat/completions`,
        {
          model: request.options.model || AIModel.GLM_4,
          messages: [
            { role: 'user', content: request.prompt }
          ],
          temperature: request.options.temperature || 0.7,
          max_tokens: request.options.maxTokens || 2000,
          top_p: request.options.topP || 0.9
        },
        {
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json'
          },
          timeout: 60000 // 60 seconds
        }
      );

      const content = response.data.choices[0].message.content;
      const tokensUsed = response.data.usage?.total_tokens || 0;
      const inputTokens = response.data.usage?.prompt_tokens || 0;
      const outputTokens = response.data.usage?.completion_tokens || 0;

      const cost = this.calculateCost(inputTokens, outputTokens, this.pricing);

      logger.info(`GLM generation successful. Tokens: ${tokensUsed}, Cost: ¥${cost.toFixed(4)}`);

      return {
        content,
        provider: AIProvider.GLM,
        model: request.options.model || AIModel.GLM_4,
        tokensUsed,
        cost,
        timestamp: new Date()
      };
    } catch (error) {
      throw this.handleError(error, AIProvider.GLM);
    }
  }

  async estimateCost(prompt: string, options: any): Promise<CostEstimate> {
    // Rough estimation: 1 token ≈ 0.75 word (Chinese) or 0.5 word (English)
    const estimatedInputTokens = Math.ceil(prompt.length * 1.5);
    const estimatedOutputTokens = options.maxTokens || 2000;

    const inputCost = this.calculateCost(estimatedInputTokens, 0, this.pricing);
    const outputCost = this.calculateCost(0, estimatedOutputTokens, this.pricing);

    return {
      estimatedTokens: estimatedInputTokens + estimatedOutputTokens,
      estimatedCost: inputCost + outputCost,
      currency: 'CNY',
      breakdown: {
        input: inputCost,
        output: outputCost
      }
    };
  }

  async validateApiKey(): Promise<boolean> {
    try {
      await axios.post(
        `${this.baseUrl}chat/completions`,
        {
          model: AIModel.GLM_4,
          messages: [{ role: 'user', content: 'test' }],
          max_tokens: 5
        },
        {
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json'
          },
          timeout: 10000
        }
      );
      return true;
    } catch (error) {
      logger.error('GLM API key validation failed:', error);
      return false;
    }
  }
}
```

```typescript
// backend/src/services/ai/ai-manager.service.ts
import { AIProvider, AIModel, TextGenerationRequest, TextGenerationResult, CostEstimate } from '../../types/ai.types';
import { GLMService } from './glm.service';
import { OpenAIService } from './openai.service';
import { GeminiService } from './gemini.service';
import { config } from '../../config';
import { logger } from '../../utils/logger';

// Fallback chain configuration
const FALLBACK_CHAIN = {
  [AIProvider.GLM]: [AIProvider.OPENAI, AIProvider.GEMINI],
  [AIProvider.OPENAI]: [AIProvider.GLM, AIProvider.GEMINI],
  [AIProvider.GEMINI]: [AIProvider.GLM, AIProvider.OPENAI]
};

export class AIServiceManager {
  private services: Map<AIProvider, any>;
  private defaultProvider: AIProvider;

  constructor() {
    this.services = new Map();
    this.initializeServices();
    this.defaultProvider = AIProvider.GLM; // Default to GLM-4
  }

  private initializeServices() {
    // Initialize GLM service
    if (config.ai.glm.apiKey) {
      this.services.set(AIProvider.GLM, new GLMService(
        config.ai.glm.apiKey,
        config.ai.glm.baseUrl
      ));
      logger.info('GLM service initialized');
    }

    // Initialize OpenAI service
    if (config.ai.openai.apiKey) {
      this.services.set(AIProvider.OPENAI, new OpenAIService(
        config.ai.openai.apiKey,
        config.ai.openai.baseUrl
      ));
      logger.info('OpenAI service initialized');
    }

    // Initialize Gemini service
    if (config.ai.gemini.apiKey) {
      this.services.set(AIProvider.GEMINI, new GeminiService(
        config.ai.gemini.apiKey,
        config.ai.gemini.baseUrl
      ));
      logger.info('Gemini service initialized');
    }
  }

  async generateText(
    request: TextGenerationRequest,
    enableFallback: boolean = true
  ): Promise<TextGenerationResult> {
    const primaryProvider = request.provider || this.defaultProvider;
    let currentProvider = primaryProvider;
    let attempts = 0;
    const maxAttempts = enableFallback ? 3 : 1;

    while (attempts < maxAttempts) {
      try {
        const service = this.services.get(currentProvider);

        if (!service) {
          throw new Error(`AI service ${currentProvider} not initialized`);
        }

        logger.info(`Attempting generation with ${currentProvider} (attempt ${attempts + 1}/${maxAttempts})`);

        const result = await service.generateText({
          ...request,
          provider: currentProvider,
          options: {
            ...request.options,
            model: request.options.model || this.getDefaultModel(currentProvider)
          }
        });

        // Log if fallback was used
        if (attempts > 0) {
          logger.info(`Successfully generated using fallback ${currentProvider} after ${attempts} attempts`);
        }

        return result;

      } catch (error: any) {
        attempts++;
        logger.error(`Generation failed with ${currentProvider}:`, error.message);

        if (!enableFallback || attempts >= maxAttempts) {
          throw error;
        }

        // Try next provider in fallback chain
        const fallbackChain = FALLBACK_CHAIN[currentProvider as AIProvider];
        if (fallbackChain && fallbackChain.length > 0) {
          currentProvider = fallbackChain[attempts % fallbackChain.length];
          logger.info(`Falling back to ${currentProvider}`);
        } else {
          throw error;
        }
      }
    }

    throw new Error('All generation attempts failed');
  }

  async estimateCost(
    prompt: string,
    provider: AIProvider,
    options: any
  ): Promise<CostEstimate> {
    const service = this.services.get(provider);

    if (!service) {
      throw new Error(`AI service ${provider} not initialized`);
    }

    return await service.estimateCost(prompt, options);
  }

  async validateAllServices(): Promise<Map<AIProvider, boolean>> {
    const results = new Map<AIProvider, boolean>();

    for (const [provider, service] of this.services) {
      try {
        const isValid = await service.validateApiKey();
        results.set(provider, isValid);
        logger.info(`${provider} API key validation: ${isValid ? 'SUCCESS' : 'FAILED'}`);
      } catch (error) {
        logger.error(`Error validating ${provider}:`, error);
        results.set(provider, false);
      }
    }

    return results;
  }

  getAvailableProviders(): AIProvider[] {
    return Array.from(this.services.keys());
  }

  private getDefaultModel(provider: AIProvider): AIModel {
    const defaults = {
      [AIProvider.GLM]: AIModel.GLM_4,
      [AIProvider.OPENAI]: AIModel.GPT_4,
      [AIProvider.GEMINI]: AIModel.GEMINI_PRO
    };
    return defaults[provider] || AIModel.GLM_4;
  }
}

// Export singleton instance
export const aiServiceManager = new AIServiceManager();
```

```typescript
const WS_PORT = parseInt(process.env.WS_PORT || '4001');

export function setupWebSocketServer() {
  const httpServer = createServer();
  const wss = new WebSocketServer({ server: httpServer });

  wss.on('connection', (ws: WebSocket, req) => {
    const clientId = generateClientId();
    logger.info(`WebSocket client connected: ${clientId}`);

    // Send welcome message
    ws.send(JSON.stringify({
      type: 'connected',
      clientId,
      message: 'Connected to Conmebution WebSocket server'
    }));

    // Handle incoming messages
    ws.on('message', (message: string) => {
      try {
        const data = JSON.parse(message);
        handleClientMessage(ws, clientId, data);
      } catch (error) {
        logger.error('Error parsing WebSocket message:', error);
        ws.send(JSON.stringify({
          type: 'error',
          message: 'Invalid message format'
        }));
      }
    });

    // Handle disconnect
    ws.on('close', () => {
      logger.info(`WebSocket client disconnected: ${clientId}`);
    });

    // Handle errors
    ws.on('error', (error) => {
      logger.error(`WebSocket error for client ${clientId}:`, error);
    });
  });

  // Start WebSocket server
  httpServer.listen(WS_PORT, () => {
    logger.info(`🔌 WebSocket server running on ws://localhost:${WS_PORT}`);
  });

  return wss;
}

function generateClientId(): string {
  return `client_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
}

function handleClientMessage(ws: WebSocket, clientId: string, data: any) {
  logger.info(`Received message from ${clientId}:`, data.type);

  switch (data.type) {
    case 'subscribe-task':
      // Subscribe to task updates
      subscribeToTask(ws, clientId, data.taskId);
      break;

    case 'unsubscribe-task':
      // Unsubscribe from task updates
      unsubscribeFromTask(ws, clientId, data.taskId);
      break;

    case 'ping':
      // Heartbeat
      ws.send(JSON.stringify({ type: 'pong' }));
      break;

    default:
      ws.send(JSON.stringify({
        type: 'error',
        message: `Unknown message type: ${data.type}`
      }));
  }
}

function subscribeToTask(ws: WebSocket, clientId: string, taskId: string) {
  logger.info(`${clientId} subscribed to task ${taskId}`);

  // Send current task status if exists
  const progress = getTaskProgress(taskId);
  if (progress) {
    ws.send(JSON.stringify({
      type: 'task-update',
      data: progress
    }));
  }

  // Watch for task updates (simple implementation)
  // In production, use Redis pub/sub for better scalability
  const interval = setInterval(() => {
    const latestProgress = getTaskProgress(taskId);

    if (!latestProgress) {
      clearInterval(interval);
      return;
    }

    ws.send(JSON.stringify({
      type: 'task-update',
      data: latestProgress
    }));

    // Stop watching if task is completed or failed
    if (latestProgress.status === 'completed' || latestProgress.status === 'failed') {
      clearInterval(interval);
    }
  }, 500); // Poll every 500ms

  // Store interval for cleanup
  (ws as any).subscriptions = (ws as any).subscriptions || {};
  (ws as any).subscriptions[taskId] = interval;
}

function unsubscribeFromTask(ws: WebSocket, clientId: string, taskId: string) {
  logger.info(`${clientId} unsubscribed from task ${taskId}`);

  const subscriptions = (ws as any).subscriptions || {};
  if (subscriptions[taskId]) {
    clearInterval(subscriptions[taskId]);
    delete subscriptions[taskId];
  }
}
```

```typescript
  } catch (error: any) {
    logger.error(`Text generation job ${job.id} failed:`, error);

    // Update content history status to failed if record exists
    if (contentHistoryId) {
      try {
        await prisma.contentHistory.update({
          where: { id: contentHistoryId },
          data: { status: 'draft' } // Keep as draft for retry
        });
      } catch (updateError) {
        logger.error('Failed to update content history status:', updateError);
      }
    }

    const errorProgress = updateTaskProgress(job.id, {
      status: TaskStatus.FAILED,
      progress: 0,
      currentStep: 'Generation failed',
      error: {
        type: error.code || 'GENERATION_ERROR',
        message: error.message || 'Failed to generate content',
        retryable: error.retryable !== false
      }
    });

    throw new Error(errorProgress.error?.message || 'Generation failed');
  }
}
```

```typescript
// frontend/lib/websocket.ts
const WS_URL = process.env.NEXT_PUBLIC_WS_URL || 'ws://localhost:4001';

export type TaskMessage = {
  type: 'connected' | 'task-update' | 'error' | 'pong';
  clientId?: string;
  data?: any;
  message?: string;
};

export class WebSocketClient {
  private ws: WebSocket | null = null;
  private clientId: string | null = null;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  private reconnectDelay = 2000;
  private messageHandlers: Map<string, (data: any) => void> = new Map();

  connect(): Promise<void> {
    return new Promise((resolve, reject) => {
      try {
        this.ws = new WebSocket(WS_URL);

        this.ws.onopen = () => {
          console.log('WebSocket connected');
          this.reconnectAttempts = 0;
          resolve();
        };

        this.ws.onmessage = (event) => {
          try {
            const message: TaskMessage = JSON.parse(event.data);
            console.log('WebSocket message received:', message);

            // Handle different message types
            switch (message.type) {
              case 'connected':
                this.clientId = message.clientId || null;
                break;

              case 'task-update':
              case 'error':
              case 'pong':
                // Call registered handler
                const handler = this.messageHandlers.get(message.type);
                if (handler) {
                  handler(message.data || message);
                }
                break;
            }
          } catch (error) {
            console.error('Error parsing WebSocket message:', error);
          }
        };

        this.ws.onerror = (error) => {
          console.error('WebSocket error:', error);
          reject(error);
        };

        this.ws.onclose = () => {
          console.log('WebSocket disconnected');
          this.attemptReconnect();
        };
      } catch (error) {
        reject(error);
      }
    });
  }

  private attemptReconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      console.log(`Attempting to reconnect (${this.reconnectAttempts}/${this.maxReconnectAttempts})...`);

      setTimeout(() => {
        this.connect().catch((error) => {
          console.error('Reconnection failed:', error);
        });
      }, this.reconnectDelay);
    } else {
      console.error('Max reconnection attempts reached');
    }
  }

  subscribeToTask(taskId: string, onUpdate: (data: any) => void) {
    if (!this.ws || this.ws.readyState !== WebSocket.OPEN) {
      console.error('WebSocket not connected');
      return;
    }

    // Register handler for task updates
    this.messageHandlers.set('task-update', onUpdate);

    // Send subscription message
    this.send({
      type: 'subscribe-task',
      taskId
    });
  }

  unsubscribeFromTask(taskId: string) {
    if (!this.ws || this.ws.readyState !== WebSocket.OPEN) {
      return;
    }

    this.send({
      type: 'unsubscribe-task',
      taskId
    });

    // Remove handler
    this.messageHandlers.delete('task-update');
  }

  send(data: any) {
    if (!this.ws || this.ws.readyState !== WebSocket.OPEN) {
      console.error('WebSocket not connected');
      return;
    }

    this.ws.send(JSON.stringify(data));
  }

  disconnect() {
    if (this.ws) {
      this.ws.close();
      this.ws = null;
    }
    this.messageHandlers.clear();
  }

  isConnected(): boolean {
    return this.ws !== null && this.ws.readyState === WebSocket.OPEN;
  }
}

// Singleton instance
let wsClient: WebSocketClient | null = null;

export function getWebSocketClient(): WebSocketClient {
  if (!wsClient) {
    wsClient = new WebSocketClient();
  }
  return wsClient;
```

```typescript
import { useState, useEffect, useRef } from 'react';
import { getWebSocketClient, TaskMessage } from '@/lib/websocket';

export interface TaskProgress {
  taskId: string;
  status: 'pending' | 'processing' | 'completed' | 'failed';
  progress: number;
  currentStep: string;
  result?: any;
  error?: {
    type: string;
    message: string;
    retryable: boolean;
  };
  startedAt: string;
  completedAt?: string;
}

export function useWebSocketTask(taskId: string | null) {
  const [progress, setProgress] = useState<TaskProgress | null>(null);
  const [isConnected, setIsConnected] = useState(false);
  const wsClientRef = useRef<typeof getWebSocketClient extends () => infer T ? T : null>(null);

  useEffect(() => {
    const client = getWebSocketClient();
    wsClientRef.current = client as any;

    // Connect to WebSocket server
    if (!client.isConnected()) {
      client.connect().then(() => {
        setIsConnected(true);
      }).catch((error) => {
        console.error('Failed to connect to WebSocket:', error);
        setIsConnected(false);
      });
    } else {
      setIsConnected(true);
    }

    // Cleanup on unmount
    return () => {
      // Don't disconnect, keep connection for other components
    };
  }, []);

  useEffect(() => {
    if (!taskId || !wsClientRef.current) return;

    const client = wsClientRef.current as any;

    // Subscribe to task updates
    client.subscribeToTask(taskId, (data: TaskProgress) => {
      console.log('Task update received:', data);
      setProgress(data);
    });

    // Cleanup
    return () => {
      client.unsubscribeFromTask(taskId);
    };
  }, [taskId]);

  return {
    progress,
    isConnected
  };
}
```

```typescript
const API_URL = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:4000';

export interface GenerationRequest {
  prompt: string;
  provider: string;
  options?: {
    temperature?: number;
    maxTokens?: number;
    topP?: number;
  };
}

export interface GenerationResponse {
  success: boolean;
  data?: {
    taskId: string;
    status: string;
    message: string;
  };
  error?: any;
}

export interface TaskStatusResponse {
  success: boolean;
  data?: any;
  error?: any;
}

export class APIClient {
  private baseUrl: string;

  constructor(baseUrl: string = API_URL) {
    this.baseUrl = baseUrl;
  }

  private async request<T>(
    endpoint: string,
    options?: RequestInit
  ): Promise<T> {
    const url = `${this.baseUrl}${endpoint}`;

    try {
      const response = await fetch(url, {
        ...options,
        headers: {
          'Content-Type': 'application/json',
          ...options?.headers
        }
      });

      const data = await response.json();

      if (!response.ok) {
        throw new Error(data.error?.message || 'Request failed');
      }

      return data;
    } catch (error) {
      console.error('API request failed:', error);
      throw error;
    }
  }

  async generateText(request: GenerationRequest): Promise<GenerationResponse> {
    return this.request<GenerationResponse>('/api/generate/text', {
      method: 'POST',
      body: JSON.stringify(request)
    });
  }

  async getTaskStatus(taskId: string): Promise<TaskStatusResponse> {
    return this.request<TaskStatusResponse>(`/api/tasks/${taskId}`);
  }

  async healthCheck(): Promise<any> {
    return this.request<any>('/api/health');
  }
}
```
